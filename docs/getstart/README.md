## 在项目中使用

### 安装

```javascript
 npm install alienupload --save-dev
 
```

### 使用

```javascript
 import ValienUpload from 'alienupload'
 Vue.use(ValienUpload)
 
```

### 按需加载

```javascript
 import ValienUpload from 'alienupload'
 components:{
      'alienUpload': ValienUpload.upload,
   },
 
```
